 $reg = new REG_api();

 $reg->addParam( "command"  "domain_modify_ns" );

 $reg->addParam( "domein", "registratiecentrale" );
 $reg->addParam( "tld",    "nl"                 );

 $reg->addParam( "ns1",    "ns1.mijndnsserver.nl" );
 $reg->addParam( "ns1_ip", "89.69.83.184"         );
 $reg->addParam( "ns2",    "ns2.mijndnsserver.nl" );
 $reg->addParam( "ns2_ip", "80.69.80.181"         );

 $reg->DoTransaction();

 if( $reg->Values[ "errcount" ] > 0 ) {
  print "Fout opgetreden";

  for( $i=1;$i<=$reg->Values[ "errcount" ];$i++ ) {
   print $reg->Values[ "errno".$i ];
  }
 }
 else {
  print "OK";
 }
