 $reg = new REG_api();

 $reg->addParam( "command", "domain_list" );
 $reg->addParam( "sort", "registrant" );
 $reg->addParam( "order", "1" );

 $reg->DoTransaction();

 if( $reg->Values[ "errcount" ] > 0 ) {
  print "Fout opgetreden";

  for( $i=1;$i<=$reg->Values[ "errcount" ];$i++ ) {
   print $reg->Values[ "errno".$i ];
  }
 }
 else {
  print "OK<BR>";
  print "<TABLE>";
   for($i=0;$i<=$reg->Values["domaincount"];$i++) {
    print "<TR>";
    print "<TD>" . $reg->Values["domain[$i]"]       . "</TD>";
    print "<TD>" . $reg->Values["registrant[$i]"]   . "</TD>";
    print "<TD>" . $reg->Values["admin[$i]"]        . "</TD>";
    print "<TD>" . $reg->Values["tech[$i]"]         . "</TD>";
    print "<TD>" . $reg->Values["verloopdatum[$i]"] . "</TD>";
    print "<TD>" . $reg->Values["status[$i]"]       . "</TD>";
    print "</TR>";
   }
  print "</TABLE>";
 }
