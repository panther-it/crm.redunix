Voorbeeld 1

 $reg = new REG_api();

 $reg->addParam( "command", "nameserver_list" );

 $reg->addParam( "sort",  "ns1" );
 $reg->addParam( "order", "1"   );

 $reg->DoTransaction();

 if( $reg->Values[ "errcount" ] > 0 ) {
  print "Fout opgetreden";

  for( $i=1;$i<=$reg->Values[ "errcount" ];$i++ ) {
   print $reg->Values[ "errno".$i ];
  }
 }
 else {
  print "OK<BR>";
  print "<TABLE>";
   for($i=0;$i<=$reg->Values[ "nscount" ];$i++) {
    print "<TR>";
    print "<TD>" . $reg->Values[ "ns_id[$i]" ]  . "</TD>";
    print "<TD>" . $reg->Values[ "ns1[$i]" ]    . "</TD>";
    print "<TD>" . $reg->Values[ "ns1_ip[$i]" ] . "</TD>";
    print "<TD>" . $reg->Values[ "ns2[$i]" ]    . "</TD>";
    print "<TD>" . $reg->Values[ "ns2_ip[$i]" ] . "</TD>";
    print "</TR>";
   }
  print "</TABLE>";
 }
