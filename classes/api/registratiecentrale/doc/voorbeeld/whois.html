 $reg = new REG_api();

 $reg->addParam( "command", "whois" );

 $reg->addParam( "type",   "uitgebreid"            );
 $reg->addParam( "domein", "registratiecentrale.nl" );

 $reg->DoTransaction();

 if( $reg->Values[ "errcount" ] > 0 ) {
  print "Fout opgetreden";

  for( $i=1;$i<=$reg->Values[ "errcount" ];$i++ ) {
   print $reg->Values[ "errno".$i ];
  }
 }
 else {
  print "OK<BR>";
  print $reg->Values[ "result" ];

 }
