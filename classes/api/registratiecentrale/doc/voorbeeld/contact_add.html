 $reg = new REG_api();

 $reg->addParam( "command", "contact_add" );

 $reg->addParam( "soort",         "full"                       );
 $reg->addParam( "bedrijfsnaam",  "De Registratiecentrale"         );
 $reg->addParam( "voorletter",    "M"                          );
 $reg->addParam( "tussenvoegsel", ""                           );
 $reg->addParam( "achternaam",    "bonsee"                     );
 $reg->addParam( "tel",           "036-8445369"                );
 $reg->addParam( "email",         "info@registratiecentrale.nl" );
 $reg->addParam( "straat",        "Lindengouw"                 );
 $reg->addParam( "huisnr",        "67"                         );
 $reg->addParam( "huisnrtoev",    ""                           );
 $reg->addParam( "postcode",      "1351 LE"                    );
 $reg->addParam( "plaats",        "Almere"                     );
 $reg->addParam( "land",          "nl"                         );

 $reg->DoTransaction();

 if( $reg->Values[ "errcount" ] > 0 ) {
  print "Fout opgetreden";

  for( $i=1;$i<=$reg->Values[ "errcount" ];$i++ ) {
   print $reg->Values[ "errno".$i ];
  }
 }
 else {
  print "OK<BR>";
  print "contact_id: " . $reg->Values[ "contact_id" ];
 }
