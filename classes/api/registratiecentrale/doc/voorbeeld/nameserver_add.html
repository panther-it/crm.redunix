 $reg = new REG_api();

 $reg->addParam( "command", "nameserver_add" );

 $reg->addParam( "auto", "false" );

 $reg->addParam( "ns1",    "ns1.mijndnsserver.nl" );
 $reg->addParam( "ns1_ip", "80.69.83.184"         );
 $reg->addParam( "ns2",    "ns1.mijndnsserver.nl" );
 $reg->addParam( "ns2_ip", "80.69.80.191"         );

 $reg->DoTransaction();

 if( $reg->Values[ "errcount" ] > 0 ) {
  print "Fout opgetreden";

  for( $i=1;$i<=$reg->Values[ "errcount" ];$i++ ) {
   print $reg->Values[ "errno".$i ];
  }
 }
 else {
  print "OK<BR>";
  print "ns_id: " . $reg->Values[ "ns_id" ];
 }
