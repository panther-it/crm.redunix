Voorbeeld: contact_list

Voorbeeld 1

 $reg = new REG_api();

 $reg->addParam( "command", "contact_list" );
 $reg->addParam( "sort", "bedrijfsnaam" );
 $reg->addParam( "order", "1" );

 $reg->DoTransaction();

 if( $reg->Values[ "errcount" ] > 0 ) {
  print "Fout opgetreden";

  for( $i=1;$i<=$reg->Values[ "errcount" ];$i++ ) {
   print $reg->Values[ "errno".$i ];
  }
 }
 else {
  print "OK<BR>";
  print "<TABLE>";
   for($i=0;$i<=$reg->Values["contactcount"];$i++) {
    print "<TR>";
    print "<TD>" . $reg->Values[ "contact_id[$i]" ]            . "</TD>";
    print "<TD>" . $reg->Values[ "contact_bedrijfsnaam[$i]" ]  . "</TD>";
    print "<TD>" . $reg->Values[ "contact_voorletter[$i]" ]    . "</TD>";
    print "<TD>" . $reg->Values[ "contact_tussenvoegsel[$i]" ] . "</TD>";
    print "<TD>" . $reg->Values[ "contact_achternaam[$i]" ]    . "</TD>";
    print "<TD>" . $reg->Values[ "contact_email[$i]" ]         . "</TD>";
    print "<TD>" . $reg->Values[ "contact_tel[$i]" ]           . "</TD>";
    print "<TD>" . $reg->Values[ "contact_soort[$i]" ]         . "</TD>";
    print "</TR>";
   }
  print "</TABLE>";
 }
